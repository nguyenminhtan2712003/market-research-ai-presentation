MỤC TIÊU:
Vào Google Sheet, tìm các keyword bị trùng trong sheet "Tiềm_năng" ở cột "Keyword", xóa các dòng trùng đi — luôn GIỮ LẠI dòng ở TRÊN, XÓA dòng ở DƯỚI.
THÔNG TIN KẾT NỐI:

- Spreadsheet ID: 161wxoOt4l...KMKEbJs
- Sheet name: Tiềm_năng
FLOW HOẠT ĐỘNG:
1. Đọc toàn bộ dữ liệu từ sheet
2. So sánh keyword (không phân biệt hoa/thường, bỏ khoảng trắng thừa)
3. In danh sách các dòng trùng để user xem trước
4. Hỏi user: gõ YES để xóa / NO để hủy
5. Nếu YES → xóa toàn bộ dòng trùng khỏi sheet
LƯU Ý QUAN TRỌNG:
- Xóa từ dòng dưới cùng lên để không bị lệch số dòng

import re
from google.oauth2 import service_account
from googleapiclient.discovery import build

SERVICE_ACCOUNT_FILE = "secrets/API_KEY.json"
SPREADSHEET_ID = "161wxoOt4l...KMKEbJs"
SHEET_NAME = "Tiềm_năng"
KEYWORD_HEADER = "Keyword"

SCOPES = ["https://www.googleapis.com/auth/spreadsheets"]

def get_sheets_service():
    credentials = service_account.Credentials.from_service_account_file(
        SERVICE_ACCOUNT_FILE,
        scopes=SCOPES,
    )

    service = build("sheets", "v4", credentials=credentials)
    return service


def quote_sheet_name(sheet_name):
    escaped = sheet_name.replace("'", "''")
    return f"'{escaped}'"


def get_sheet_id_by_name(service, spreadsheet_id, sheet_name):
    spreadsheet = service.spreadsheets().get(
        spreadsheetId=spreadsheet_id
    ).execute()

    sheets = spreadsheet.get("sheets", [])

    for sheet in sheets:
        properties = sheet.get("properties", {})
        if properties.get("title") == sheet_name:
            return properties.get("sheetId")

    raise ValueError(f"Không tìm thấy tab sheet tên: {sheet_name}")


def normalize_text(value):
    if value is None:
        return ""

    text = str(value).strip().lower()
    text = re.sub(r"\s+", " ", text)

    return text


def get_all_rows(service, spreadsheet_id, sheet_name):
    range_name = f"{quote_sheet_name(sheet_name)}!A:Z"

    result = service.spreadsheets().values().get(
        spreadsheetId=spreadsheet_id,
        range=range_name,
    ).execute()

    return result.get("values", [])


def find_keyword_column_index(header_row, keyword_header):
    target = normalize_text(keyword_header)

    for index, cell in enumerate(header_row):
        if normalize_text(cell) == target:
            return index

    raise ValueError(f'Không tìm thấy cột header tên "{keyword_header}"')


def find_duplicate_keyword_rows(rows, keyword_column_index):
    seen_keywords = {}
    duplicate_rows = []

    for row_index in range(1, len(rows)):
        row = rows[row_index]

        if keyword_column_index >= len(row):
            continue

        raw_keyword = row[keyword_column_index]
        normalized_keyword = normalize_text(raw_keyword)

        if not normalized_keyword:
            continue

        actual_row_number = row_index + 1

        if normalized_keyword in seen_keywords:
            duplicate_rows.append({
                "row_number": actual_row_number,
                "keyword": raw_keyword,
                "duplicate_of_row": seen_keywords[normalized_keyword]["row_number"],
                "duplicate_of_keyword": seen_keywords[normalized_keyword]["keyword"],
            })
        else:
            seen_keywords[normalized_keyword] = {
                "row_number": actual_row_number,
                "keyword": raw_keyword,
            }

    return duplicate_rows


def delete_rows(service, spreadsheet_id, sheet_id, row_numbers):
    if not row_numbers:
        print("Không có dòng trùng để xóa.")
        return

    # Xóa từ dưới lên để không bị lệch số dòng
    sorted_rows = sorted(row_numbers, reverse=True)

    requests = []

    for row_number in sorted_rows:
        requests.append({
            "deleteDimension": {
                "range": {
                    "sheetId": sheet_id,
                    "dimension": "ROWS",
                    "startIndex": row_number - 1,
                    "endIndex": row_number,
                }
            }
        })

    body = {
        "requests": requests
    }

    service.spreadsheets().batchUpdate(
        spreadsheetId=spreadsheet_id,
        body=body,
    ).execute()

    print(f"Đã xóa {len(sorted_rows)} dòng trùng.")
    print("Các dòng đã xóa:", sorted_rows)


def preview_duplicates():
    service = get_sheets_service()

    rows = get_all_rows(service, SPREADSHEET_ID, SHEET_NAME)

    if not rows:
        print("Sheet đang trống.")
        return []

    header_row = rows[0]
    keyword_column_index = find_keyword_column_index(header_row, KEYWORD_HEADER)

    duplicates = find_duplicate_keyword_rows(rows, keyword_column_index)

    print("===== PREVIEW DUPLICATE KEYWORDS =====")
    print(f"Sheet: {SHEET_NAME}")
    print(f"Cột Keyword: {KEYWORD_HEADER}")
    print(f"Tổng số dòng: {len(rows)}")
    print(f"Số dòng keyword trùng: {len(duplicates)}")
    print()

    if duplicates:
        for item in duplicates:
            print(
                f"Dòng {item['row_number']} | "
                f"Keyword: {item['keyword']} | "
                f"Trùng với dòng {item['duplicate_of_row']}"
            )
    else:
        print("Không có keyword trùng.")

    return duplicates


def dedupe_keywords():
    service = get_sheets_service()

    rows = get_all_rows(service, SPREADSHEET_ID, SHEET_NAME)

    if not rows:
        print("Sheet đang trống.")
        return

    sheet_id = get_sheet_id_by_name(service, SPREADSHEET_ID, SHEET_NAME)

    header_row = rows[0]
    keyword_column_index = find_keyword_column_index(header_row, KEYWORD_HEADER)

    duplicates = find_duplicate_keyword_rows(rows, keyword_column_index)

    duplicate_row_numbers = [item["row_number"] for item in duplicates]

    print("===== DELETE DUPLICATE KEYWORDS =====")
    print(f"Sheet: {SHEET_NAME}")
    print(f"Số dòng keyword trùng sẽ xóa: {len(duplicate_row_numbers)}")

    delete_rows(
        service=service,
        spreadsheet_id=SPREADSHEET_ID,
        sheet_id=sheet_id,
        row_numbers=duplicate_row_numbers,
    )


if __name__ == "__main__":
    # Mặc định chỉ preview, chưa xóa
    preview_duplicates()

    while True:
        confirm = input(
            "\nBạn có muốn xóa các dòng trùng không? Gõ YES để xóa / Gõ NO để hủy: "
        ).strip().upper()

        if confirm == "YES":
            dedupe_keywords()
            break
        elif confirm == "NO":
            print("Đã hủy. Chưa xóa dòng nào.")
            break
        else:
            print("Bạn gõ sai so với yêu cầu. Vui lòng chỉ gõ YES hoặc NO.")

Prompt → AI viết code Check Trademark Keyword

💬 Prompt gửi AI Cowork
Viết script Python làm việc này:
MỤC TIÊU:
Đọc keyword từ Google Sheet, nhờ Gemini AI đánh giá rủi ro trademark, rồi ghi kết quả vào cột "Check TM".
THÔNG TIN KẾT NỐI:
- Xác thực Google Sheets bằng Service Account JSON tại "secrets/API_KEY.json"
- Spreadsheet ID: 161wxoOt4l...KMKEbJs
- Sheet: Tiềm_năng | Cột kết quả: "Check TM"
- Nếu cột "Check TM" chưa có thì tự tạo thêm
FLOW HOẠT ĐỘNG:
1. Đọc các keyword chưa có giá trị ở cột Check TM
2. Chia thành batch 100 keyword, gửi lên Gemini để hỏi
3. Gemini trả về: keyword nào rủi ro thì WARNING, không thì SAFE
4. Ghi WARNING vào cột Check TM — keyword SAFE thì để trống
5. In tổng kết cuối: bao nhiêu WARNING / SAFE / lỗi
CÁCH GỬI GEMINI:
- Dùng Selenium mở Chrome, vào gemini.google.com
- Chrome lưu profile tại "chrome_profiles/gemini" để giữ đăng nhập
- Dùng pyperclip paste prompt vào ô chat (tránh lỗi ký tự)
- Mỗi batch mở chat mới, đợi tối đa 120s lấy kết quả
- Kết quả Gemini trả về nằm giữa tag BEGIN_RESULT và END_RESULT
TIÊU CHÍ GEMINI ĐÁNH GIÁ WARNING:
Thương hiệu, tên nhân vật, ca sĩ/diễn viên nổi tiếng,
franchise game/anime/phim, hoặc thuộc class 16 trademarkia

def normalize_text(value):
    if value is None:
        return ""
    text = str(value).strip().lower()
    text = re.sub(r"\s+", " ", text)
    return text


def strip_markdown(text):
    """Loại bỏ ký tự markdown mà Gemini hay thêm vào: **bold**, *italic*, `code`."""
    text = re.sub(r"\*{1,3}(.+?)\*{1,3}", r"\1", text)   # **bold** / *italic*
    text = re.sub(r"`(.+?)`", r"\1", text)                 # `code`
    text = re.sub(r"_{1,2}(.+?)_{1,2}", r"\1", text)       # _italic_ / __bold__
    return text.strip()


def column_index_to_letter(index_zero_based):
    number  = index_zero_based + 1
    letters = ""
    while number:
        number, remainder = divmod(number - 1, 26)
        letters = chr(65 + remainder) + letters
    return letters


def get_all_rows(service, spreadsheet_id, sheet_name):

def find_header_column_index(header_row, header_name):
    target = normalize_text(header_name)
    for index, cell in enumerate(header_row):
        if normalize_text(cell) == target:
            return index
    return -1


def ensure_header_column(service, spreadsheet_id, sheet_name, header_name):
    rows = get_all_rows(service, spreadsheet_id, sheet_name)

    if not rows:
        raise ValueError(f'Sheet "{sheet_name}" đang trống hoặc chưa có header.')

    header_row     = rows[0]
    existing_index = find_header_column_index(header_row, header_name)

    if existing_index != -1:
        return existing_index

    new_col_index  = len(header_row)
    new_col_letter = column_index_to_letter(new_col_index)

    service.spreadsheets().values().update(
        spreadsheetId=spreadsheet_id,
        range=f"{quote_sheet_name(sheet_name)}!{new_col_letter}1",
        valueInputOption="USER_ENTERED",
        body={"values": [[header_name]]},
    ).execute()

    print(f'Đã tạo thêm cột "{header_name}" tại cột {new_col_letter}.')
    return new_col_index


def get_unchecked_keywords(service, spreadsheet_id, sheet_name):
    """Lấy keywords chưa có giá trị trong cột Check TM."""
    rows = get_all_rows(service, spreadsheet_id, sheet_name)

    if not rows:
        raise ValueError(f'Sheet "{sheet_name}" đang trống.')

    header_row = rows[0]

    keyword_col_index = find_header_column_index(header_row, KEYWORD_HEADER)
    if keyword_col_index == -1:
        raise ValueError(f'Không tìm thấy cột "{KEYWORD_HEADER}".')

    check_tm_col_index = ensure_header_column(
        service, spreadsheet_id, sheet_name, CHECK_TM_HEADER
    )

    keywords = []

    for row_index in range(1, len(rows)):
        row     = rows[row_index]
        keyword = row[keyword_col_index] if keyword_col_index < len(row) else ""
        keyword = str(keyword).strip()

        if not keyword:
            continue

        # Bỏ qua keyword đã có giá trị trong cột Check TM
        check_tm_val = row[check_tm_col_index] if check_tm_col_index < len(row) else ""
        if str(check_tm_val).strip():
            continue

        keywords.append({
            "row_number":         row_index + 1,
            "keyword":            keyword,
            "check_tm_col_index": check_tm_col_index,
        })

    return keywords


def update_check_tm_values(service, spreadsheet_id, sheet_name, keyword_items, result_map):
    updates = []

    for item in keyword_items:
        keyword            = item["keyword"]
        row_number         = item["row_number"]
        check_tm_col_index = item["check_tm_col_index"]
        normalized_keyword = normalize_text(keyword)

        status = result_map.get(normalized_keyword)

        if status is None:
            print(f'  [!] Gemini bỏ sót: "{keyword}" → ghi WARNING để review.')
            status = "WARNING"

        write_value = WRITE_WARNING_VALUE if status == "WARNING" else WRITE_SAFE_VALUE

        # Nếu WRITE_SAFE_VALUE = "" thì bỏ qua dòng SAFE (không ghi gì)
        if write_value == "":
            continue

        col_letter = column_index_to_letter(check_tm_col_index)
        updates.append({
            "range":  f"{quote_sheet_name(sheet_name)}!{col_letter}{row_number}",
            "values": [[write_value]],
        })

    if not updates:
        print("  Không có dữ liệu Check TM nào để cập nhật.")
        return

    service.spreadsheets().values().batchUpdate(
        spreadsheetId=spreadsheet_id,
        body={"valueInputOption": "USER_ENTERED", "data": updates},
    ).execute()

    print(f"  ✓ Đã cập nhật {len(updates)} dòng vào cột Check TM.")

def create_chrome_driver():
    os.makedirs(CHROME_PROFILE_DIR, exist_ok=True)

    options = Options()
    options.add_argument("--start-maximized")
    options.add_argument("--lang=en-US")
    options.add_argument(f"--user-data-dir={CHROME_PROFILE_DIR}")

    driver = webdriver.Chrome(options=options)
    driver.set_page_load_timeout(60)
    return driver


def wait_for_gemini_ready(driver):

def build_tm_prompt(keyword_items):
    keyword_lines = "\n".join(f"- {item['keyword']}" for item in keyword_items)

    return f"""Đây là bộ từ khóa chính:

{keyword_lines}

HÃY TỰ LOẠI BỎ NHỮNG TỪ KHÓA DÍNH ĐẾN THƯƠNG HIỆU, TÊN NHÂN VẬT, TÊN CA SĨ/DIỄN VIÊN NỔI TIẾNG CÓ TRADEMARK.

Yêu cầu:
1. Review khả năng bị đánh bản quyền / trademark / thương hiệu / nhân vật / người nổi tiếng của từng keyword.
2. Keyword nào có rủi ro trademark, brand name, celebrity, character, copyrighted franchise, team, movie/game/anime/music reference/ hoặc thuộc class 16 trong trademarkia thì để Check TM là WARNING.
3. Keyword nào không thấy rủi ro rõ ràng thì để Check TM là SAFE.
4. Giữ nguyên chính xác từng keyword gốc, không được sửa chính tả hay thêm bớt ký tự.
5. Chỉ trả về bảng Markdown nằm giữa BEGIN_RESULT và END_RESULT.
6. Không giải thích thêm ngoài bảng.

Format bắt buộc:

BEGIN_RESULT
| Keyword | Check TM |
|---|---|
| keyword gốc | SAFE hoặc WARNING |
END_RESULT""".strip()


def find_gemini_input_box(driver):
    for selector in ["div[contenteditable='true']", "textarea"]:
        for element in driver.find_elements(By.CSS_SELECTOR, selector):
            if element.is_displayed() and element.is_enabled():
                return element
    raise RuntimeError("Không tìm thấy ô nhập prompt của Gemini.")


def click_gemini_send_button(driver):
    for selector in [
        'button[aria-label*="Send"]',
        'button[aria-label*="Gửi"]',
        'button[aria-label*="Submit"]',
        'button[data-test-id*="send"]',
    ]:
        for button in driver.find_elements(By.CSS_SELECTOR, selector):
            if button.is_displayed() and button.is_enabled():
                button.click()
                return True
    return False


def send_prompt_to_gemini(driver, prompt):
    input_box = find_gemini_input_box(driver)
    input_box.click()
    time.sleep(1)

    pyperclip.copy(prompt)
    input_box.send_keys(Keys.CONTROL, "v")
    time.sleep(1)

    sent = click_gemini_send_button(driver)
    if not sent:
        input_box.send_keys(Keys.ENTER)

    print("  Đã gửi prompt sang Gemini.")


def wait_for_gemini_result(driver, timeout_seconds=RESPONSE_TIMEOUT_SECONDS):
    print(f"  Đang chờ Gemini trả kết quả (tối đa {timeout_seconds}s)...")

    try:
        WebDriverWait(driver, timeout_seconds).until(
            lambda d: "BEGIN_RESULT" in d.find_element(By.TAG_NAME, "body").text
            and "END_RESULT" in d.find_element(By.TAG_NAME, "body").text
        )
    except Exception:
        body_text = driver.find_element(By.TAG_NAME, "body").text
        print(f"  [DEBUG] Timeout hoặc lỗi. Nội dung trang (200 ký tự cuối):")
        print(f"  {body_text[-200:]}")
        raise RuntimeError(
            f"Gemini không trả kết quả trong {timeout_seconds}s. "
            "Thử tăng RESPONSE_TIMEOUT_SECONDS trong config."
        )

    time.sleep(2)

    body_text = driver.find_element(By.TAG_NAME, "body").text

    result_blocks = re.findall(
        r"BEGIN_RESULT\s*(.*?)\s*END_RESULT",
        body_text,
        flags=re.DOTALL | re.IGNORECASE,
    )

    if not result_blocks:
        print("  [DEBUG] Không tìm thấy block. Nội dung trang (500 ký tự cuối):")
        print(f"  {body_text[-500:]}")
        raise RuntimeError("Không tìm thấy block BEGIN_RESULT / END_RESULT trong kết quả Gemini.")

    return result_blocks[-1].strip()


def _try_match(norm_key, original_lookup):
    """Tìm key trong original_lookup: chính xác trước, fuzzy sau."""
    if norm_key in original_lookup:
        return norm_key
    for orig_norm in original_lookup:
        if orig_norm in norm_key or norm_key in orig_norm:
            return orig_norm
    return None


def parse_markdown_table_result(result_block, keyword_items):
    """
    Parse kết quả Gemini — hỗ trợ 2 format:
      1. Bảng Markdown : | keyword | SAFE |
      2. Plain text    : keyword SAFE
    """
    original_lookup = {
        normalize_text(item["keyword"]): item["keyword"]
        for item in keyword_items
    }

    result_map = {}
    skipped    = []

    for line in result_block.splitlines():
        line = line.strip()
        if not line:
            continue

        # ── Format 1: Markdown table ──────────────────────────────
        if line.startswith("|"):
            # Bỏ separator |---|---|
            if re.search(r"^\|\s*[-:]+\s*\|\s*[-:]+\s*\|?$", line):
                continue

            columns = [col.strip() for col in line.strip("|").split("|")]
            if len(columns) < 2:
                continue

            raw_keyword = strip_markdown(columns[0])
            status      = columns[1].strip().upper()

            if normalize_text(raw_keyword) in ("keyword", "check tm", "từ khóa", ""):
                continue
            if status not in ("SAFE", "WARNING"):
                continue

            matched = _try_match(normalize_text(raw_keyword), original_lookup)
            if matched:
                result_map[matched] = status
            else:
                skipped.append(raw_keyword)

        # ── Format 2: Plain text "keyword SAFE" / "keyword WARNING" ─
        else:
            # Dòng kết thúc bằng SAFE hoặc WARNING (có thể có dấu chấm, dấu phẩy)
            m = re.match(
                r"^(.+?)\s+(SAFE|WARNING)[.,;]?\s*$",
                line,
                flags=re.IGNORECASE,
            )
            if not m:
                continue

            raw_keyword = strip_markdown(m.group(1).strip())
            status      = m.group(2).upper()

            if normalize_text(raw_keyword) in ("keyword", "check tm", "từ khóa", ""):
                continue

            matched = _try_match(normalize_text(raw_keyword), original_lookup)
            if matched:
                result_map[matched] = status
            else:
                skipped.append(raw_keyword)

    if skipped:
        print(f"  [DEBUG] {len(skipped)} dòng không khớp keyword gốc:")
        for kw in skipped[:10]:
            print(f"    - \"{kw}\"")

    return result_map

def run_task_kw_b3():
    print("===== TASKKWB3: CHECK TRADEMARK BY GEMINI WEB =====")

    service = get_sheets_service()

    # Đọc toàn bộ keyword chưa check
    all_keywords = get_unchecked_keywords(service, SPREADSHEET_ID, SOURCE_SHEET_NAME)

    print(f'Sheet nguồn: "{SOURCE_SHEET_NAME}"')
    print(f"Tổng keyword chưa check: {len(all_keywords)}")

    if not all_keywords:
        print("Không có keyword nào cần check. Kết thúc.")
        return

    # Chia batch
    batches = [
        all_keywords[i:i + BATCH_SIZE]
        for i in range(0, len(all_keywords), BATCH_SIZE)
    ]
    total_batches = len(batches)
    print(f"Sẽ xử lý {total_batches} batch (mỗi batch ≤ {BATCH_SIZE} keywords).")

    driver = None

    try:
        driver = create_chrome_driver()
        wait_for_gemini_ready(driver)

        total_warning = 0
        total_safe    = 0
        total_error   = 0

        for batch_index, batch_items in enumerate(batches, start=1):
            print()
            print(f"{'─' * 50}")
            print(f"BATCH {batch_index}/{total_batches} — {len(batch_items)} keywords")
            print(f"{'─' * 50}")

            # Mở conversation mới cho mỗi batch (trừ batch đầu)
            if batch_index > 1:
                open_new_gemini_chat(driver)

            prompt = build_tm_prompt(batch_items)

            try:
                send_prompt_to_gemini(driver, prompt)
                result_block = wait_for_gemini_result(driver)

                print()
                print("  ── KẾT QUẢ RAW TỪ GEMINI ──")
                print(result_block)
                print()

                result_map = parse_markdown_table_result(result_block, batch_items)
                print(f"  Parse được {len(result_map)}/{len(batch_items)} keyword.")

                if not result_map:
                    print("  [!] Không parse được kết quả — bỏ qua batch này.")
                    total_error += len(batch_items)
                    continue

                # Đếm warning/safe
                for status in result_map.values():
                    if status == "WARNING":
                        total_warning += 1
                    else:
                        total_safe += 1

                update_check_tm_values(
                    service, SPREADSHEET_ID, SOURCE_SHEET_NAME, batch_items, result_map
                )

            except Exception as error:
                print(f"  [!] Lỗi batch {batch_index}: {error}")
                print("  → Bỏ qua batch này, tiếp tục batch sau.")
                total_error += len(batch_items)
                continue

        print()
        print("=" * 50)
        print("  HOÀN TẤT TASKKWB3")
        print(f"  WARNING : {total_warning}")
        print(f"  SAFE    : {total_safe}")
        print(f"  Lỗi/bỏ sót : {total_error}")
        print("=" * 50)

    finally:
        if driver:
            driver.quit()
            print("Đã tắt Chrome Gemini.")


if __name__ == "__main__":
    run_task_kw_b3()
